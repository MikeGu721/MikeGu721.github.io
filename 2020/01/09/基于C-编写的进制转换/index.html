比较有意思的时，这里写的进制转换，不限与八进制十进制，十进制以下几进制转换都能进行

第一部分是最正常的进制转换

代码将分成两步：

1.将P进制数x转换为十进制y：

​ 这步的思想是将x的第n位乘以P的n次方累加到y

2.将十进制数y转换为Q进制数z

​ 这步的思想是将y除以Q，得到的余数存于低位，并将商继续重复前面的过程

第二部分是自己编写的一个输入两个十进制数输出这两个数在D进制下的和的函数

#include <iostream>//将P进制的x转化为Q进制数输出
int main()
{
    int x, P, product = 1, y = 0;//P是输入进制数，x是输入数，product用于计算P的n次方，y是十进制数
    printf("请输入数字：");
    scanf_s("%d", &x);
    printf("请输入进制数：");
    scanf_s("%d", &P);
    while (x != 0) {
        y = y + (x % 10) * product;
        x = x / 10;
        product = product * P;
    }//将P进制数x转换为十进制y

int z[40], num = 0, Q;//Q是输出进制数，z是输出数，num用来指示数组
printf("请输入输出进制数：");
scanf_s("%d", &Q);
do {
    z[num++] = y % Q;
    y = y / Q;
} while (y != 0);//将十进制数y转换为Q进制数z
printf("输出的数字为：");
for (int i = num - 1; i >= 0; i--) {
    printf("%d", z[i]);
}
//________________________________________________
//下面的功能是输入十进制a和b，输出D进制的和
int D, a, b;
printf("\n下面进入add函数调试模块\n");
printf("add函数功能是输入两个十进制数，输出他们的D进制数的和\n");
printf("请输入进制数：");
scanf_s("%d", &D);
printf("请输入两个十进制加数：");
scanf_s("%d%d", &a, &b);
printf("输出%d进制和为：", D);
int all = a + b;
int out[40], i;
for (i = 0; all!=0; i++) {
    out[i] = all % D;
    all = all / D;
}
for (int p = i - 1; p >= 0; p--) {
    printf("%d", out[p]);
}

}
